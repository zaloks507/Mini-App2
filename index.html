let users = []; let messages = []; let chatBox = document.getElementById('chat-box'); let input = document.getElementById('message-input'); let sendBtn = document.getElementById('send-btn'); let userList = document.getElementById('user-list'); // Функция авторизации через Telegram function onTelegramAuth(user) { // user = {id, first_name, last_name, username, photo_url} console.log("Telegram user:", user); // Скрываем виджет, показываем приложение document.getElementById('login-container').style.display = 'none'; document.getElementById('app').style.display = 'block'; // Добавляем текущего пользователя в список let currentUser = { id: user.id, name: user.first_name + (user.last_name ? ' ' + user.last_name : ''), username: user.username, avatar: user.photo_url, status: 'активен', messages: 0 }; users.push(currentUser); // Отображаем список renderUsers(); renderChat(); // Сохраняем текущего пользователя window.currentUser = currentUser; } // Рендер списка пользователей function renderUsers() { userList.innerHTML = ''; users.forEach(user => { let btn = document.createElement('button'); btn.textContent = user.name; btn.onclick = () => showProfile(user.id); userList.appendChild(btn); }); } // Рендер чата function renderChat() { chatBox.innerHTML = ''; messages.forEach(msg => { const div = document.createElement('div'); div.textContent = `${msg.user}: ${msg.text}`; chatBox.appendChild(div); }); chatBox.scrollTop = chatBox.scrollHeight; } // Отправка сообщения sendBtn.addEventListener('click', () => { const text = input.value.trim(); if (!text) return; messages.push({user: window.currentUser.name, text}); input.value = ''; renderChat(); // Увеличиваем счетчик сообщений window.currentUser.messages++; renderUsers(); }); // Профиль function showProfile(id) { const user = users.find(u => u.id === id); if (!user) return; document.getElementById('profile-avatar').src = user.avatar || 'https://via.placeholder.com/120'; document.getElementById('profile-name').textContent = user.name; document.getElementById('profile-status').textContent = `Статус: ${user.status}`; document.getElementById('profile-messages').textContent = `Сообщений: ${user.messages}`; document.getElementById('profile').style.display = 'block'; } function closeProfile() { document.getElementById('profile').style.display = 'none'; }
